# structural_x3dna
Process MD data by x3dna

# Create a virtual environment
`conda create --name x3dna python=3.8`

# Activate virtual environment
`conda activate x3dna`

# Deactivate
`conda deactivate`

# Install structural_x3dna package
`pip install -e .`

# Upgrade package
`pip install -e . --upgrade`
